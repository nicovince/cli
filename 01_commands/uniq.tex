\subsection{uniq}

\begin{frame}[fragile]{uniq}
\begin{itemize}
\item Suppress identical adjacent lines.
\pause \item Often used after a \emph{sort} operation.
\end{itemize}
  \pause
  \begin{exampleblock}{}
    \begin{lstlisting}[showstringspaces=false]
> echo -e "a\na" | uniq
a
    \end{lstlisting}
  \end{exampleblock}
\end{frame}
